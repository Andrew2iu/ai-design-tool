#### 方案A：添加Ollama本地化支持（推荐，工作量中等）

**1) 修改 `backend/python/app/core/config.py`**
```python
import os
from dotenv import load_dotenv

load_dotenv()

# --- 多模型配置 ---
AI_PROVIDER = os.getenv("AI_PROVIDER", "deepseek")  # deepseek / ollama / openai

# DeepSeek 云端
DEEPSEEK_API_KEY = os.getenv("DEEPSEEK_API_KEY", "")
DEEPSEEK_BASE_URL = os.getenv("DEEPSEEK_BASE_URL", "https://api.deepseek.com")
DEEPSEEK_MODEL = os.getenv("DEEPSEEK_MODEL", "deepseek-chat")

# Ollama 本地
OLLAMA_BASE_URL = os.getenv("OLLAMA_BASE_URL", "http://localhost:11434/v1")
OLLAMA_MODEL = os.getenv("OLLAMA_MODEL", "deepseek-r1")  # 或其他本地模型

# OpenAI GPT
OPENAI_API_KEY = os.getenv("OPENAI_API_KEY", "")
OPENAI_BASE_URL = os.getenv("OPENAI_BASE_URL", "https://api.openai.com/v1")
OPENAI_MODEL = os.getenv("OPENAI_MODEL", "gpt-4o")

def get_ai_client():
    """根据AI_PROVIDER返回对应的OpenAI兼容客户端"""
    from openai import OpenAI
    
    if AI_PROVIDER == "ollama":
        return OpenAI(api_key="ollama", base_url=OLLAMA_BASE_URL), OLLAMA_MODEL
    elif AI_PROVIDER == "openai":
        return OpenAI(api_key=OPENAI_API_KEY, base_url=OPENAI_BASE_URL), OPENAI_MODEL
    else:  # deepseek
        return OpenAI(api_key=DEEPSEEK_API_KEY, base_url=DEEPSEEK_BASE_URL), DEEPSEEK_MODEL
```

```python
from app.core.config import get_ai_client

_client, _model = None, None

def _get_client():
    global _client, _model
    if _client is None:
        _client, _model = get_ai_client()
    return _client, _model
```

```python
# backend/python/app/skills/base.py
from abc import ABC, abstractmethod
from typing import Dict, Any

class BaseSkill(ABC):
    """Anima Skill范式基础类"""
    name: str
    description: str
    design_aware: bool = True  # 是否感知设计语言
    
    @abstractmethod
    async def execute(self, context: Dict[str, Any], params: Dict[str, Any]) -> Dict[str, Any]:
        pass

# backend/python/app/skills/registry.py
class SkillRegistry:
    _skills: Dict[str, BaseSkill] = {}
    
    def register(self, skill: BaseSkill):
        self._skills[skill.name] = skill
    
    def get(self, name: str) -> BaseSkill:
        return self._skills.get(name)
    
    def list_all(self) -> list:
        return [{"name": s.name, "description": s.description, "design_aware": s.design_aware} for s in self._skills.values()]

# 具体Skill实现
class DesignGenerationSkill(BaseSkill):
    name = "design_generation"
    description = "自然语言→设计稿生成"
    design_aware = True
    
    async def execute(self, context, params):
        prompt = params.get("prompt", "")
        design_system = context.get("design_system", {})
        # 调用AI服务生成设计稿...
        return {"design_json": ...}

class BrandConsistencySkill(BaseSkill):
    name = "brand_consistency"
    description = "设计规范约束与品牌一致性检查"
    design_aware = True
    
    async def execute(self, context, params):
        design = params.get("design", {})
        # 合规检查...
        return {"compliance_report": ...}

class CodeGenerationSkill(BaseSkill):
    name = "code_generation"
    description = "设计稿→代码转换"
    design_aware = True
    
    async def execute(self, context, params):
        design = params.get("design", {})
        framework = params.get("framework", "react")
        # 代码生成...
        return {"code": ...}
```

## 五、风险提示

1. **Ollama硬件要求**：本地运行DeepSeek-R1需要至少16GB内存/GPU，学生机器可能不够用，建议docker-compose中注释掉GPU配置，默认用云端DeepSeek
2. **MDUI版本兼容**：MDUI v2基于Web Components，与React集成需注意shadow DOM问题，建议仅在设计Token层面集成，组件渲染仍用自定义方案
3. **模拟数据→真实数据**：MCP Server从模拟改为真实后，需要FastAPI有对应的"获取当前画布状态"API，当前routes.py中没有此端点，需新增
